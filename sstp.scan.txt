#!/data/data/com.termux/files/usr/bin/bash

RED="\033[31m"
GREEN="\033[32m"
ORANGE="\033[33m"
CYAN="\033[36m"
MAGENTA="\033[35m"
RESET="\033[0m"

TAG="${MAGENTA}┌────────────────────┐
│ telegram👉@ShadowProxy66 │
└────────────────────┘${RESET}"

# ----------------------------------------------------------
# GeoIP Country Lookup
# ----------------------------------------------------------
get_country() {
    ip="$1"
    curl -fsSL "http://ip-api.com/json/$ip" | jq -r '.country // "Unknown"'
}

# ----------------------------------------------------------
# Option 1 — SSTP Scanner
# ----------------------------------------------------------
scan_sstp() {
    PAGE_URL="https://ipspeed.info/free-sstp.php"

    echo "[*] Fetching SSTP servers..."
    HTML="$(curl -fsSL "$PAGE_URL")" || { echo "[!] Fetch failed"; return; }

    mapfile -t RAW_HOSTS < <(echo "$HTML" | grep -Eo '[a-zA-Z0-9.-]+\.[a-z]{2,6}:[0-9]{2,5}')

    if [[ ${#RAW_HOSTS[@]} -eq 0 ]]; then
      echo "[!] No hosts found"
      return
    fi

    echo "[*] Testing servers..."
    for item in "${RAW_HOSTS[@]}"; do
      host="${item%%:*}"
      port="${item##*:}"
      [[ "$host" == "$port" ]] && port=443

      country=$(get_country "$host")

      if timeout 5 bash -c "echo > /dev/tcp/${host}/${port}" 2>/dev/null; then
        echo -e "${GREEN}[OPEN] ${host}:${port} (${country})${RESET}\n$TAG"
      else
        echo -e "${RED}[CLOSED] ${host}:${port} (${country})${RESET}\n$TAG"
      fi
    done
}

# ----------------------------------------------------------
# Option 2 — DoH Scanner (Your New Script)
# ----------------------------------------------------------
scan_doh() {

PAGE_URL="https://github.com/curl/curl/wiki/DNS-over-HTTPS#publicly-available-servers"
RESULTS="$HOME/.cache/doh_results.txt"

rm -f "$RESULTS"

echo "[+] Fetching DoH list..."
RAW=$(curl -sL "$PAGE_URL")

DNS_LIST=($(printf '%s\n' "$RAW" | grep -oE 'https://[^ \"'\''()]+dns-query' | sort -u))
TOTAL=${#DNS_LIST[@]}

if [ "$TOTAL" -eq 0 ]; then
    echo "[!] No DoH servers found!"
    return
fi

echo "[+] Running parallel latency tests..."
BEST_DNS=""
BEST_TIME=999999
COUNT=0

for DNS in "${DNS_LIST[@]}"; do
(
    START=$(date +%s%3N)
    curl -s --http2 --max-time 2 -H 'accept: application/dns-json' "${DNS}?name=example.com&type=A" > /dev/null
    END=$(date +%s%3N)
    LATENCY=$((END - START))

    echo "$DNS $LATENCY" >> "$RESULTS"
) &
done

# Live progress percentage
while [ "$COUNT" -lt "$TOTAL" ]; do
    COUNT=$(wc -l < "$RESULTS" 2>/dev/null)
    PERCENT=$((COUNT * 100 / TOTAL))
    echo -ne "🔄 Progress: ${PERCENT}% \r"
    sleep 0.1
done

echo ""

# Determine fastest DoH
while read -r DNS LAT; do
    if [ "$LAT" -lt "$BEST_TIME" ] && [ "$LAT" -gt 0 ]; then
        BEST_TIME=$LAT
        BEST_DNS=$DNS
    fi
done < "$RESULTS"

echo "=============================="
echo -e "🔥 Fastest DNS-over-HTTPS server:"
echo -e "➡️ ${GREEN}$BEST_DNS${RESET}"
echo -e "⏱ Latency: ${GREEN}$BEST_TIME ms${RESET}"
echo "=============================="

rm -f "$RESULTS"
}

# ----------------------------------------------------------
# MENU
# ----------------------------------------------------------
while true; do
    clear
    echo -e "${MAGENTA}======================================${RESET}"
    echo -e "${MAGENTA}         Hamid Scanner Menu           ${RESET}"
    echo -e "${MAGENTA}======================================${RESET}"
    echo
    echo -e "${ORANGE}1) Scan SSTP${RESET}"
    echo -e "${CYAN}2) Scan DoH DNS${RESET}"
    echo -e "${CYAN}0) Exit${RESET}"
    echo
    read -p "Choose option: " opt

    case $opt in
        1) scan_sstp; read -p "Press Enter...";;
        2) scan_doh; read -p "Press Enter...";;
        0) exit ;;
        *) echo "Invalid choice"; sleep 1 ;;
    esac
done
