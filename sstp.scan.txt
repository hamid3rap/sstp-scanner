#!/usr/bin/env bash

GREEN="\033[32m"
RED="\033[31m"
CYAN="\033[36m"
RESET="\033[0m"

echo -e "${CYAN}[+] Fetching SSTP servers from ipspeed.info...${RESET}"

# Fetch SSTP links from the website
RAW=$(curl -s https://ipspeed.info)

# Extract SSTP links
LIST=$(echo "$RAW" | grep -Eo 'sstp://[A-Za-z0-9\.\-_:]+' | sort -u)

if [[ -z "$LIST" ]]; then
    echo -e "${RED}[!] No SSTP links found.${RESET}"
    exit 1
fi

echo -e "${CYAN}[+] Found $(echo "$LIST" | wc -l) SSTP servers${RESET}"
echo ""

# Function to get country name
get_country() {
    curl -s "https://ipapi.co/$1/country_name/"
}

# Process each SSTP server
echo "$LIST" | while read -r LINK; do
    HOSTPORT=$(echo "$LINK" | sed 's/sstp:\/\///')
    HOST=$(echo "$HOSTPORT" | cut -d: -f1)
    PORT=$(echo "$HOSTPORT" | cut -d: -f2)

    # Test port
    timeout 1 bash -c "echo > /dev/tcp/$HOST/$PORT" 2>/dev/null
    STATUS=$?

    COUNTRY=$(get_country "$HOST")

    echo "----------------------------------------"
    echo " SSTP Server:"
    echo " $LINK"
    echo " Country: $COUNTRY"
    echo " Port: $PORT"

    if [[ $STATUS -eq 0 ]]; then
        echo -e " Status: ${GREEN}OPEN (Working)${RESET}"
    else
        echo -e " Status: ${RED}CLOSED${RESET}"
    fi

    echo " https://t.me/ShadowProxy66"
    echo "----------------------------------------"
    echo ""
done
